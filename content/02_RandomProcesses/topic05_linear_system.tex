\begin{kbox}{5. 随机过程通过线性系统}
    设系统冲激响应为 $h(t)$，传输函数为 $H(f)$。
    \begin{itemize}
        \item \textbf{输出均值}：
        \[ E[\xi_o(t)] = \mu_{\xi} \cdot H(0) \quad \text{(直流增益)} \]
        \item \textbf{输出 PSD}：
        \[ P_{\xi o}(f) = |H(f)|^2 P_{\xi i}(f) \]
        \item \textbf{输出自相关}：
        \[ R_o(\tau) = R_i(\tau) * h(\tau) * h(-\tau) \]
    \end{itemize}
\end{kbox}
